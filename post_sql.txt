create table question_post(
    ident int auto_increment primary key, 
    title varchar(100), 
    detail varchar(1000),
    user_id INT,
    is_resolved BOOLEAN DEFAULT FALSE,
    question_time datetime,
    answer_count INT DEFAULT 0,
    FOREIGN KEY (user_id) REFERENCES account(user_id)   
);
